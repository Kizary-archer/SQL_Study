use bd_estate_agency_tr
go 
select город.название AS "Город",avg(квартира.цена_кв_м * квартира.жилплощадь) AS "Средняя цена за однокомнатную квартиру"
from город,квартира,дом,район
where квартира.дом = дом.id_дома and
	  дом.район = район.номер_района and 
	  район.город = город.номер_города
	  and квартира.кол_комнат = 1
group by город.название
